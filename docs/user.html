        if(!attr[prop]) {
          throw Error('Missing ' + prop);
        }
        
        if(attr[prop].constructor !== type) {
          throw Error(prop + ' must be a ' + type.name);
        }       
      });

        if (options[attr]) this[attr] = options[attr];
